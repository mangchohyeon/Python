#순열

from itertools import *
data = ['A','B','C']
a = list(permutations(data,2)) # a라는 리스트에 2개씩 data의 요소를 가져와서 소괄호 형태로 저장(자리바꿈 있음)

# 결과값
# [('A', 'B'), ('A', 'C'), ('B', 'A'), ('B', 'C'), ('C', 'A'), ('C', 'B')]
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
#조합

from itertools import *
data = ['A','B','C']
a = list(combinations(data,2)) # a라는 리스트에 2개씩 data의 요소를 가져와서 소괄호 형태로 저장(자리바꿈 없음)

# 결과값
# [('A', 'B'), ('A', 'C'), ('B', 'C')]
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

#중복순열(순열에다가 중복 포함)

data = ['A', 'B', 'C'] 
a = list(product(data, repeat = 2))
print(a)

# 결과값
# [('A', 'A'), ('A', 'B'), ('A', 'C'), ('B', 'A'), ('B', 'B'), ('B', 'C'), ('C', 'A'), ('C', 'B'), ('C', 'C')]
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
